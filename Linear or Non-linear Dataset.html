Here's a basic implementation of linear weighted regression in Python using NumPy. In weighted linear regression, different data points can have different weights, and the model fits those points accordingly.

Code for Linear Weighted Regression:

python
import numpy as np
import matplotlib.pyplot as plt

# Sample data (replace these with your actual data)
X = np.array([1, 2, 3, 4, 5])  # Feature
y = np.array([1, 2, 1.3, 3.75, 2.25])  # Target values
weights = np.array([1, 1, 2, 1, 3])  # Weights for each data point

# Adding a column of ones to X to account for the bias term (intercept)
X_b = np.c_[np.ones((len(X), 1)), X]

# Calculate the weighted normal equation solution for linear regression
W = np.diag(weights)  # Creating a diagonal matrix of weights
theta = np.linalg.inv(X_b.T.dot(W).dot(X_b)).dot(X_b.T).dot(W).dot(y)

# Predicting the values
y_pred = X_b.dot(theta)

# Plotting the data points and the regression line
plt.scatter(X, y, color="blue", label="Data points")
plt.plot(X, y_pred, color="red", label="Weighted Regression Line")
plt.xlabel("X")
plt.ylabel("y")
plt.title("Linear Weighted Regression")
plt.legend()
plt.show()

# Print the parameters
print(f"Intercept (Bias): {theta[0]}")
print(f"Coefficient (Slope): {theta[1]}")

Explanation:

  1. X and y: These are your features (input) and target (output) values.
  2. weights: This array contains the weights for each data point, where higher weights give more importance to the corresponding points.
  3. X_b: A column of ones is added to the features for the intercept term.
  4. W: A diagonal matrix is created from the weights.
  5. theta: This is the solution to the normal equation, modified for weighted data.

You can replace the X, y, and weights arrays with your actual dataset and desired weights.

x = data[x_col].values
y = data[y_col].values

# Function to calculate weights for each point
def get_weights(query_point, X, tau):
    # tau is the bandwidth parameter (larger tau means smoother curve)
    m = X.shape[0]
    W = np.mat(np.eye(m))
    for i in range(m):
        xi = X[i]
        W[i, i] = np.exp(np.dot((xi - query_point), (xi - query_point).T) / (-2.0 * tau ** 2))
    return W

# Locally Weighted Linear Regression implementation
def lwlr(query_point, X, Y, tau):
    X = np.mat(X)
    Y = np.mat(Y).T  # Transpose Y to make it a column vector
    m = X.shape[0]
    
    # Add a bias term (column of 1s) to X
    X_b = np.hstack((np.ones((m, 1)), X))
    
    # Get the weights for the query point
    W = get_weights(query_point, X, tau)
    
    # Apply the formula:  = (XWX) XWY
    theta = (X_b.T * W * X_b).I * (X_b.T * W * Y)
    
    # Predict the output for the query point
    query_point_b = np.mat([1, query_point])  # Add bias term for query point
    prediction = query_point_b * theta
    return prediction[0, 0]

# Plotting Locally Weighted Linear Regression fit
def plot_lwlr(X, Y, tau):
    X_copy = X.copy()
    X_sorted = np.sort(X_copy)  # Sort X values to get a smoother plot
    
    # Apply LWLR for each point in X
    y_pred = np.array([lwlr(xi, X, Y, tau) for xi in X_sorted])
    
    plt.scatter(X, Y, alpha=0.5, label='Data Points')  # Plot the original data points
    plt.plot(X_sorted, y_pred, color='red', label=f'LWLR Fit (tau={tau})', linewidth=2)  # Plot the LWLR curve
    plt.xlabel(x_col)
    plt.ylabel(y_col)
